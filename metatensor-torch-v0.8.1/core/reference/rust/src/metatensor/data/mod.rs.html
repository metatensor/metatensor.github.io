20    #[test]
21    fn shape() {
22        let array = Box::new(ArrayD::from_elem(vec![3, 4, 2], 1.0)) as Box<dyn Array>;
23        let mut array = unsafe { ArrayRefMut::new(array.into()) };
24
25        assert_eq!(array.as_raw().shape().unwrap(), [3, 4, 2]);
26        array.as_raw_mut().reshape(&[12, 2]).unwrap();
27        assert_eq!(array.as_raw().shape().unwrap(), [12, 2]);
28        assert_eq!(array.as_array(), ArrayD::from_elem(vec![12, 2], 1.0));
29
30        array.as_raw_mut().swap_axes(0, 1).unwrap();
31        assert_eq!(array.as_raw().shape().unwrap(), [2, 12]);
32    }

34    #[test]
35    fn create() {
36        let array = Box::new(ArrayD::from_elem(vec![4, 2], 1.0)) as Box<dyn Array>;
37        let array = unsafe { ArrayRef::from_raw(array.into()) };
38
39        assert_eq!(get_data_origin(array.as_raw().origin().unwrap()).unwrap(), "rust.Box<dyn Array>");
40        assert_eq!(array.as_array(), ArrayD::from_elem(vec![4, 2], 1.0));
41
42        let other = unsafe { ArrayRef::from_raw(array.as_raw().create(&[5, 3, 7, 12]).unwrap()) };
43        assert_eq!(other.as_raw().shape().unwrap(), [5, 3, 7, 12]);
44        assert_eq!(get_data_origin(other.as_raw().origin().unwrap()).unwrap(), "rust.Box<dyn Array>");
45        assert_eq!(other.as_array(), ArrayD::from_elem(vec![5, 3, 7, 12], 0.0));
46    }

48    #[test]
49    fn move_samples_from() {
50        let array = Box::new(ArrayD::from_elem(vec![3, 2, 2, 4], 1.0)) as Box<dyn Array>;
51        let array = unsafe { ArrayRef::from_raw(array.into()) };
52
53        let mut other = unsafe { ArrayRefMut::new(array.as_raw().create(&[1, 2, 2, 8]).unwrap()) };
54        assert_eq!(other.as_array(), ArrayD::from_elem(vec![1, 2, 2, 8], 0.0));
55
56        let mapping = mts_sample_mapping_t {
57            output: 0,
58            input: 1,
59        };
60        other.as_raw_mut().move_samples_from(array.as_raw(), &[mapping], 2..6).unwrap();
61        let expected = ArrayD::from_shape_vec(vec![1, 2, 2, 8], vec![
62                0.0, 0.0, 1.0, 1.0, 1.0, 1.0, 0.0, 0.0,
63                0.0, 0.0, 1.0, 1.0, 1.0, 1.0, 0.0, 0.0,
64                0.0, 0.0, 1.0, 1.0, 1.0, 1.0, 0.0, 0.0,
65                0.0, 0.0, 1.0, 1.0, 1.0, 1.0, 0.0, 0.0,
66        ]).unwrap();
67        assert_eq!(other.as_array(), expected);
68    }
